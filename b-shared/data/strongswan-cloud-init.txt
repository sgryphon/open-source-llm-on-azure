#cloud-config
# strongSwan IKEv2 road-warrior VPN gateway.
#
# This file is a template. `b-shared/05-Deploy-StrongSwanVm.ps1` substitutes
# every `#INIT_*#` token before passing the result to `az vm create
# --custom-data`. If you see a `#INIT_*#` token at boot time, the render step
# failed its post-render assertion; check the PowerShell script output.
#
# The VM fetches its cert material from Key Vault via its system-assigned
# managed identity at first boot -- nothing private is substituted into this
# file by the PowerShell script. EAP credentials are the only exception; they
# are injected into /etc/swanctl/conf.d/secrets.conf below (root-read-only).

apt:
  conf: |
    Acquire::Retries "60";
    DPkg::Lock::Timeout "60";

package_upgrade: true
packages:
  - strongswan
  - strongswan-swanctl
  - libcharon-extra-plugins
  - libstrongswan-extra-plugins
  - iptables-persistent
  - azure-cli
  - ufw

write_files:
  # OS-level IP forwarding (Azure NIC-level forwarding is enabled by the
  # PowerShell script via `az network nic update --ip-forwarding true`).
  - path: /etc/sysctl.d/99-strongswan.conf
    owner: root:root
    permissions: '0644'
    content: |
      net.ipv4.ip_forward = 1
      net.ipv6.conf.all.forwarding = 1
      net.ipv6.conf.default.forwarding = 1

  # swanctl connection: one IKEv2 road-warrior with TWO remote auth options
  # (EAP-MSCHAPv2 and client-cert pubkey). The server always authenticates
  # with its cert (mutual auth). Cert material is fetched from Key Vault by
  # the first runcmd step below before `swanctl --load-all` runs.
  - path: /etc/swanctl/swanctl.conf
    owner: root:root
    permissions: '0644'
    content: |
      connections {
        roadwarrior {
          version = 2
          proposals = aes256-sha256-modp2048,default
          pools = vpn_pool_ipv4, vpn_pool_ipv6
          dpd_delay = 30s
          send_cert = always
          unique = never
          local {
            auth = pubkey
            certs = server.pem
            id = #INIT_SERVER_FQDNS#
          }
          remote-eap {
            auth = eap-mschapv2
            eap_id = %any
          }
          remote-pubkey {
            auth = pubkey
            cacerts = ca.pem
          }
          children {
            ikev2-roadwarrior {
              local_ts = 0.0.0.0/0,::/0
              esp_proposals = aes256-sha256,default
              dpd_action = clear
              start_action = none
            }
          }
        }
      }

      pools {
        vpn_pool_ipv4 {
          addrs = #INIT_VIP_POOL_IPV4#
          dns = 168.63.129.16
        }
        vpn_pool_ipv6 {
          addrs = #INIT_VIP_POOL_IPV6#
        }
      }

  # EAP credential (plaintext; root-only).
  # strongSwan also reads pubkey auth from certs under /etc/swanctl/{x509,x509ca}
  # so there is no shared-secret entry for the CA / client certs here.
  - path: /etc/swanctl/conf.d/secrets.conf
    owner: root:root
    permissions: '0600'
    content: |
      secrets {
        eap-#INIT_VPN_USERNAME# {
          id = #INIT_VPN_USERNAME#
          secret = "#INIT_VPN_PASSWORD#"
        }
      }

  # First-boot bootstrap script. Pulls CA + server cert + server key from
  # Key Vault via managed identity, with a retry loop to absorb RBAC
  # propagation delay, then configures firewall/NAT and loads strongSwan.
  - path: /usr/local/sbin/strongswan-bootstrap.sh
    owner: root:root
    permissions: '0755'
    content: |
      #!/usr/bin/env bash
      set -euo pipefail

      KV_NAME='#INIT_KEY_VAULT_NAME#'
      CA_SECRET='#INIT_CA_SECRET_NAME#'
      SRV_CERT_SECRET='#INIT_SERVER_CERT_SECRET_NAME#'
      SRV_KEY_SECRET='#INIT_SERVER_KEY_SECRET_NAME#'

      VPN_SUBNET_V4='#INIT_VPN_SUBNET_IPV4#'
      VPN_SUBNET_V6='#INIT_VPN_SUBNET_IPV6#'

      mkdir -p /etc/swanctl/x509ca /etc/swanctl/x509 /etc/swanctl/private

      # 1. Log in with the VM's managed identity and fetch cert material.
      # Retry for up to ~120s to cope with RBAC propagation delay.
      deadline=$((SECONDS + 120))
      until az login --identity --output none 2>/dev/null; do
        if [ $SECONDS -ge $deadline ]; then
          echo "ERROR: az login --identity did not succeed within 120s" >&2
          exit 1
        fi
        sleep 5
      done

      fetch_secret() {
        local name="$1" dest="$2" mode="$3"
        local deadline=$((SECONDS + 120))
        until az keyvault secret download --vault-name "$KV_NAME" --name "$name" --file "$dest" --output none 2>/dev/null; do
          if [ $SECONDS -ge $deadline ]; then
            echo "ERROR: failed to download Key Vault secret '$name' within 120s" >&2
            exit 1
          fi
          sleep 5
        done
        chmod "$mode" "$dest"
      }

      fetch_secret "$CA_SECRET"       /etc/swanctl/x509ca/ca.pem     644
      fetch_secret "$SRV_CERT_SECRET" /etc/swanctl/x509/server.pem   644
      fetch_secret "$SRV_KEY_SECRET"  /etc/swanctl/private/server.key 600

      # 2. Firewall / NAT rules.
      # IPv4: MASQUERADE the VPN subnet out eth0 (VNet resources see the VM's IPv4).
      # IPv6: FORWARD only; no NAT66 (ULA is routable via VNet peering).
      iptables  -t nat -C POSTROUTING -s "$VPN_SUBNET_V4" -o eth0 -j MASQUERADE 2>/dev/null || \
        iptables  -t nat -A POSTROUTING -s "$VPN_SUBNET_V4" -o eth0 -j MASQUERADE
      iptables  -C FORWARD -s "$VPN_SUBNET_V4" -j ACCEPT 2>/dev/null || \
        iptables  -A FORWARD -s "$VPN_SUBNET_V4" -j ACCEPT
      iptables  -C FORWARD -d "$VPN_SUBNET_V4" -j ACCEPT 2>/dev/null || \
        iptables  -A FORWARD -d "$VPN_SUBNET_V4" -j ACCEPT

      ip6tables -C FORWARD -s "$VPN_SUBNET_V6" -j ACCEPT 2>/dev/null || \
        ip6tables -A FORWARD -s "$VPN_SUBNET_V6" -j ACCEPT
      ip6tables -C FORWARD -d "$VPN_SUBNET_V6" -j ACCEPT 2>/dev/null || \
        ip6tables -A FORWARD -d "$VPN_SUBNET_V6" -j ACCEPT

      netfilter-persistent save

      # 3. UFW ingress (IKE + NAT-T + SSH).
      ufw allow 22/tcp
      ufw allow 500/udp
      ufw allow 4500/udp
      ufw --force enable

      # 4. Apply sysctl and start strongSwan.
      sysctl --system

      systemctl enable --now strongswan
      # Brief settle, then load config. `swanctl --load-all` is idempotent.
      sleep 2
      swanctl --load-all

      # Emit the loaded connection list for debuggability in cloud-init-output.log.
      swanctl --list-conns || true

runcmd:
  - [ /usr/local/sbin/strongswan-bootstrap.sh ]
