#cloud-config
# vLLM OpenAI-compatible inference server on a single Ubuntu 22.04 VM.
#
# The API key is baked into the cloud-init template by the PowerShell script --
# nothing private is fetched from Key Vault at runtime.

# The model bytes are NOT downloaded by cloud-init: they are placed onto
# the persistent data disk later.

package_update: true
package_upgrade: false
packages:
  - certbot
  - python3-venv
  - python3-pip
  - libcap2-bin
  - jq
  - curl
  - e2fsprogs

write_files:
  # Certbot deploy hook: copies renewed cert material into /etc/vllm/certs and restarts vllm.service.
  - path: /etc/letsencrypt/renewal-hooks/deploy/10-vllm-restart.sh
    owner: root:root
    permissions: '0755'
    content: |
      #!/bin/sh
      SOURCE_DIR=/etc/letsencrypt/live/vllm-cert
      CERTIFICATE_DIR=/etc/vllm/certs
      if [ "${RENEWED_LINEAGE}" = "${SOURCE_DIR}" ]; then
          mkdir -p "${CERTIFICATE_DIR}"
          cp "${RENEWED_LINEAGE}/fullchain.pem" "${CERTIFICATE_DIR}/server.pem"
          cp "${RENEWED_LINEAGE}/privkey.pem"   "${CERTIFICATE_DIR}/server.key"
          chown vllm:vllm "${CERTIFICATE_DIR}/server.pem" "${CERTIFICATE_DIR}/server.key"
          chmod 0600     "${CERTIFICATE_DIR}/server.pem" "${CERTIFICATE_DIR}/server.key"
          systemctl restart vllm
      fi

  # systemd unit for vLLM. ConditionPathExists keeps the unit inactive
  # (silently, no error) until the model files are loaded.
  - path: /etc/systemd/system/vllm.service
    owner: root:root
    permissions: '0644'
    content: |
      [Unit]
      Description=vLLM OpenAI-compatible server
      After=network-online.target
      Wants=network-online.target
      ConditionPathExists=#INIT_MODEL_MOUNT_POINT#/#INIT_MODEL_DIR_NAME#/config.json

      [Service]
      User=vllm
      Group=vllm
      EnvironmentFile=/etc/vllm/vllm.env
      ExecStart=/opt/vllm/.venv/bin/vllm serve \
          #INIT_MODEL_MOUNT_POINT#/#INIT_MODEL_DIR_NAME# \
          --host :: \
          --port 443 \
          --ssl-certfile /etc/vllm/certs/server.pem \
          --ssl-keyfile  /etc/vllm/certs/server.key \
          --api-key      ${VLLM_API_KEY} \
          --served-model-name #INIT_SERVED_MODEL_NAME# \
          --tool-call-parser hermes \
          --enable-auto-tool-choice \
          --max-model-len 32768
      Restart=on-failure
      RestartSec=10

      [Install]
      WantedBy=multi-user.target

  # First-boot bootstrap script.
  - path: /usr/local/sbin/vllm-bootstrap.sh
    owner: root:root
    permissions: '0755'
    content: |
      #!/usr/bin/env bash
      set -euo pipefail

      VLLM_API_KEY='#INIT_API_KEY#'
      VLLM_VERSION='#INIT_VLLM_VERSION#'
      MODEL_MOUNT='#INIT_MODEL_MOUNT_POINT#'
      HOST_NAMES='#INIT_HOST_NAMES#'
      CERT_EMAIL='#INIT_CERT_EMAIL#'

      # 1. Find the data disk and prepare /opt/models.
      #
      # Azure attaches data disks at LUN 0 /dev/disk/azure/scsi1/lun0
      # (the symlink is provided by waagent's udev rules). Wait up to ~60s
      # for the symlink to appear -- it can lag a fresh `az vm create`
      # slightly.
      DEVICE_LINK=/dev/disk/azure/scsi1/lun0
      deadline=$((SECONDS + 60))
      until [ -e "$DEVICE_LINK" ]; do
        if [ $SECONDS -ge $deadline ]; then
          echo "ERROR: data disk symlink $DEVICE_LINK did not appear within 60s" >&2
          exit 1
        fi
        sleep 2
      done
      DEVICE=$(readlink -f "$DEVICE_LINK")
      echo "data disk resolved to $DEVICE"

      # mkfs only if there is no existing filesystem. blkid prints the
      # detected type to stdout and exits 0 when something is present;
      # exits 2 with empty output when the device is blank. This is the
      # critical guard that lets the same disk survive VM rebuilds.
      if ! blkid -o value -s TYPE "$DEVICE" >/dev/null 2>&1; then
        echo "no filesystem detected on $DEVICE; running mkfs.ext4 -L llm-models"
        mkfs.ext4 -L llm-models -F "$DEVICE"
      else
        echo "existing filesystem detected on $DEVICE ($(blkid -o value -s TYPE "$DEVICE")); skipping mkfs"
      fi

      mkdir -p "$MODEL_MOUNT"
      # Persist the mount via LABEL so we are not coupled to LUN ordering
      # or device naming. `nofail` lets the VM boot if the disk is missing.
      if ! grep -qE '^LABEL=llm-models[[:space:]]' /etc/fstab; then
        echo "LABEL=llm-models $MODEL_MOUNT ext4 defaults,nofail 0 2" >> /etc/fstab
      fi
      mount -a
      echo "$MODEL_MOUNT mounted:"
      findmnt "$MODEL_MOUNT" || true

      # 2. Create the `vllm` system user (idempotent) and hand it the mount.
      if ! id -u vllm >/dev/null 2>&1; then
        useradd --system --create-home --home-dir /var/lib/vllm --shell /usr/sbin/nologin vllm
      fi
      chown vllm:vllm "$MODEL_MOUNT"

      # 3. Write /etc/vllm/vllm.env.
      mkdir -p /etc/vllm /etc/vllm/certs
      chown vllm:vllm /etc/vllm /etc/vllm/certs

      install -m 0600 -o vllm -g vllm /dev/null /etc/vllm/vllm.env
      printf 'VLLM_API_KEY=%s\nHF_HOME=%s/.cache\n' "$VLLM_API_KEY" "$MODEL_MOUNT" \
          > /etc/vllm/vllm.env

      # 4. Create the venv and install vLLM (pinned). The pin matters --
      # vLLM's CLI flags (notably --tool-call-parser and quantisation
      # options) drift between minor versions.
      python3 -m venv /opt/vllm/.venv
      /opt/vllm/.venv/bin/pip install --upgrade pip
      /opt/vllm/.venv/bin/pip install "vllm==${VLLM_VERSION}"
      chown -R vllm:vllm /opt/vllm

      # 5. setcap on the venv's python so vllm.service can bind :443
      # without running as root. The capability is set on the venv-
      # specific binary so it does not grant the privilege system-wide.
      VENV_PY=$(readlink -f /opt/vllm/.venv/bin/python)
      setcap 'cap_net_bind_service=+ep' "$VENV_PY"

      # 6. Issue the initial Let's Encrypt cert.
      # certbot standalone binds :80 transiently; vLLM never binds 80,
      # and the systemd unit is not started yet, so there is no port
      # conflict. ACME_STAGING_FLAG is empty for production, --test-cert
      # for staging.
      certbot certonly --standalone --preferred-challenges http \
          --cert-name vllm-cert \
          -d "$HOST_NAMES" \
          -n --agree-tos -m "$CERT_EMAIL"

      # 7. Run the deploy hook explicitly so the initial cert lands in
      # /etc/vllm/certs without waiting for the first renewal. The hook
      # also calls `systemctl restart vllm`, which is a no-op while the
      # unit's ConditionPathExists is unsatisfied (the model is not
      # present yet) -- exactly what we want.
      RENEWED_LINEAGE=/etc/letsencrypt/live/vllm-cert \
          /etc/letsencrypt/renewal-hooks/deploy/10-vllm-restart.sh

      # 8. Enable the unit; do NOT start it. The operator's
      # util/Import-LlmModel.ps1 will start it once the model
      # files are on disk.
      systemctl daemon-reload
      systemctl enable vllm

      echo "vllm-bootstrap complete; vllm.service enabled (inactive until model is loaded)"

runcmd:
  - [ /usr/local/sbin/vllm-bootstrap.sh ]
